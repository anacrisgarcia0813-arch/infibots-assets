import express from "express"; import cors from "cors"; import dotenv from "dotenv"; import OpenAI from "openai"; import PptxGenJS from "pptxgenjs"; dotenv.config(); const app = express(); app.use(cors()); app.use(express.json()); const openai = new OpenAI({ apiKey: process.env.OPENAI_API_KEY, }); app.post("/generate", async (req, res) => { const { topic } = req.body; try { const response = await openai.chat.completions.create({ model: "gpt-4.1", response_format: { type: "json_object" }, messages: [ { role: "system", content: ` Eres un consultor estratégico senior. Tono: - Corporativo - Analítico - Profesional - Sin lenguaje comercial agresivo Genera presentaciones ejecutivas. ` }, { role: "user", content: ` Crea una presentación sobre: ${topic} Devuelve JSON con esta estructura: { "slides":[ { "title":"", "content":"", "notes":"" } ] } Reglas: - Máximo 8 slides - Máximo 20 palabras por slide - Enfoque en estrategia, decisiones y sostenibilidad ` } ], }); const data = JSON.parse(response.choices[0].message.content); let pptx = new PptxGenJS(); data.slides.forEach(slide => { let s = pptx.addSlide(); s.addText(slide.title, { x: 0.5, y: 0.5, fontSize: 24, bold: true, }); s.addText(slide.content, { x: 0.5, y: 1.5, fontSize: 16, }); }); const fileName = `presentacion-${Date.now()}.pptx`; await pptx.writeFile({ fileName }); res.download(fileName); } catch (error) { console.error(error); res.status(500).send("Error generando presentación"); } }); app.listen(3000, () => { console.log("Servidor en http://localhost:3000"); });
